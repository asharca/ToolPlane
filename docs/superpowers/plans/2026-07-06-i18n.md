**Goal:** 让 ToolPlane 前端所有静态字段支持中英切换，MCP skill/server/client 原始描述除外。

**Architecture:** 复用现有 `next-intl` 基础设施；新增一次性 TypeScript AST 提取脚本扫描硬编码 UI 文案；人工 review 后把确认文案写入 `messages/en.json` + `zh.json`，并在源码中用 `useTranslations` / `getTranslations` 替换；最后用现有 `i18n-completeness.test.ts` 强制双语 key 一致。

**Tech Stack:** Next.js 16, React 19, next-intl 4.x, TypeScript Compiler API, Vitest.

- [ ] **Step 2: 实现最小提取函数**

```typescript
// scripts/extract-i18n-candidates.ts
import ts from 'typescript';

export interface Candidate {
  file: string;
  line: number;
  text: string;
  type: 'jsxText' | 'jsxAttribute' | 'stringLiteral';
  attrName?: string;
}

const TEXT_ATTRIBUTES = new Set([
  'label',
  'placeholder',
  'title',
  'aria-label',
  'aria-labelledby',
  'aria-describedby',
  'pendingLabel',
]);

function isMeaningfulText(text: string): boolean {
  return /[a-zA-Z一-龥]/.test(text) && text.trim().length > 0;
}

export function extractCandidatesFromSource(file: string, source: string): Candidate[] {
  const candidates: Candidate[] = [];
  const sourceFile = ts.createSourceFile(
    file,
    source,
    ts.ScriptTarget.Latest,
    true,
    ts.ScriptKind.TSX
  );

  function line(node: ts.Node): number {
    return sourceFile.getLineAndCharacterOfPosition(node.getStart()).line + 1;
  }

  function visit(node: ts.Node) {
    if (ts.isJsxText(node)) {
      const text = node.text.trim();
      if (isMeaningfulText(text)) {
        candidates.push({ file, line: line(node), text, type: 'jsxText' });
      }
    }
    if (ts.isJsxAttribute(node)) {
      const attrName = node.name.getText(sourceFile);
      if (TEXT_ATTRIBUTES.has(attrName) && node.initializer) {
        let text: string | undefined;
        if (ts.isStringLiteral(node.initializer)) {
          text = node.initializer.text;
        } else if (ts.isJsxExpression(node.initializer) && ts.isStringLiteral(node.initializer.expression)) {
          text = node.initializer.expression.text;
        }
        if (text && isMeaningfulText(text)) {
          candidates.push({ file, line: line(node), text, type: 'jsxAttribute', attrName });
        }
      }
    }
    if (ts.isStringLiteral(node)) {
      // Catch top-level string literals used as content (e.g. {'foo'})
      const parent = node.parent;
      if (ts.isJsxExpression(parent) && ts.isJsxElement(parent.parent) && isMeaningfulText(node.text)) {
        candidates.push({ file, line: line(node), text: node.text, type: 'stringLiteral' });
      }
    }
    ts.forEachChild(node, visit);
  }

  visit(sourceFile);
  return candidates;
}

export function extractCandidatesFromFiles(files: string[], readFile = ts.sys.readFile): Candidate[] {
  return files.flatMap((file) => {
    const content = readFile(file, 'utf8');
    if (!content) return [];
    return extractCandidatesFromSource(file, content);
  });
}

// CLI entry point
if (import.meta.url === `file://${process.argv[1]}`) {
  const files = process.argv.slice(2);
  if (files.length === 0) {
    console.error('Usage: pnpm tsx scripts/extract-i18n-candidates.ts <file...>');
    process.exit(1);
  }
  const candidates = extractCandidatesFromFiles(files);
  console.log(JSON.stringify(candidates, null, 2));
}
```
